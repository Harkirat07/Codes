#include<bits/stdc++.h>
using namespace std;

int max(int x,int y)
{
    return(x>y ? x:y);
}

int maxpath(int a[],int b[],int m, int n)
{
    int i=0, j=0;
    int result =0, sum1=0,sum2 =0;
    while(i<m && j<n)
    {
        if(a[i]< b[j])
        {
            sum1 += a[i++];
        }
        else if(a[i]> b[j])
        {
            sum2 += b[j++];
        }
        else
        {
            result += max(sum1,sum2);
            sum1=0;
            sum2=0;
            while(i<m && j<n && a[i]==b[j])
            {
                result+=a[i++];
                j++;
            }
        }
    }
        sum1=0;
        sum2=0;
        while(i<m)
            sum1+=a[i++];
        while(j<n)
            sum2+=a[j++];
        result+= max(sum1,sum2);
        return result;
}
int main()
{
    int ar1[]  = {2, 3, 7, 10, 12, 15, 30, 34};
    int ar2[] =  {1, 5, 7, 8, 10, 15, 16, 19};
    int m = sizeof(ar1)/sizeof(ar1[0]);
    int n = sizeof(ar2)/sizeof(ar2[0]);
    cout << "Maximum sum path is "
         << maxpath(ar1, ar2, m, n);
    return 0;
}
